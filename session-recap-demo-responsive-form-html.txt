    <script>
      const form = document.getElementById('visaForm');
      form.addEventListener('submit', function(e) {
        e.preventDefault();
        debugger;
        let isValid = true;
        const inputs = document.querySelectorAll('.val-input');
        inputs.forEach(input => {
          const icon = input.parentElement.querySelector('.text-red-500');
          if (input.value.trim() === "") {
            isValid = false;
            input.classList.add('error-shake'); // Triggers Red Border & Shake
            if (icon) icon.classList.remove('hidden');
            // Reset animation so it can re-trigger on next click
            setTimeout(() => {
              input.style.animation = 'none';
              input.offsetHeight; /* Trigger reflow */
              input.style.animation = '';
            }, 400);
          } else {
            input.classList.remove('error-shake');
            if (icon) icon.classList.add('hidden');
          }
        });
        const terms = document.getElementById('terms');
        const termsBox = document.getElementById('termsBox');
        if (!terms.checked) {
          isValid = false;
          termsBox.classList.add('error-shake');
          setTimeout(() => termsBox.classList.remove('error-shake'), 400);
        } else {
          termsBox.classList.remove('error-shake');
        }
        if (isValid) {
          alert("🎉 SUCCESS: 20-Field Application Validated!");
          form.reset();
        }
      });
    </script>
